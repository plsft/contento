Build a plugin
plugin.js 
// name: "my-awesome-plugin"
// version: "1.0.0"
// hooks: ["post:render"]

(function () {
  contento.on('post:render', function (ctx) {
    var doc = ctx.document;
    var body = doc.body;

    // Full DOM API available
    var el = doc.createElement('div');
    el.setAttribute('class', 'my-widget');
    el.textContent = 'Hello!';
    body.appendChild(el);

    // Access post metadata
    var title = ctx.title;
    var tags  = ctx.tags;
    var slug  = ctx.slug;

    // Inject styles
    var style = doc.createElement('style');
    style.textContent = '.my-widget {'
      + '  position: fixed;'
      + '  bottom: 20px;'
      + '}';
    doc.head.appendChild(style);
  });
})();
